# include<iostream>
# include<string>

const int StudentNum = 2;
const int SubjectNum = 2;

using namespace std;

double StuAvg(double score[StudentNum][SubjectNum],int num);
double ProAVG(double score[StudentNum][SubjectNum],int num);
double CompAvg(double score[StudentNum][SubjectNum],int num);

int main(){

    double Score[StudentNum][SubjectNum];

    for(int i=0;i<StudentNum;i++){
        cout << "Please input two scores for student#" << i <<" :\n";
        for(int j=0;j<SubjectNum;j++){
            cin >> Score[i][j];
        }
    }

    cout << "The average score of each student is:" << endl;
    for(int i=0;i<2;i++){
        double a = StuAvg(Score,i);
        cout << "student#" << i << " : " << a << endl;
    }

    cout << "The average score of programming is " << ProAVG(Score,0) << endl;
    cout << "The average score of basic computer concepts is " << CompAvg(Score,1);



    return 0;
}


double StuAvg(double score[StudentNum][SubjectNum],int num){
    double stdsvg=0;

    for(int j = 0;j < SubjectNum;j++){
        stdsvg += score[num][j];
    }
    stdsvg /= SubjectNum;

    return stdsvg;
}


double ProAVG(double score[StudentNum][SubjectNum],int num){
    double prosvg=0;

    for(int j = 0;j < StudentNum;j++){
        prosvg += score[j][num];
    }
    prosvg /= SubjectNum;

    return prosvg;
}


double CompAvg(double score[StudentNum][SubjectNum],int num){
    double compsvg=0;

    for(int j = 0;j < StudentNum;j++){
        compsvg += score[j][num];
    }
    compsvg /= SubjectNum;

    return compsvg;
}
