# include<iostream>
# include<string>

const int NumOfStores = 3;
const int NumOfTypes = 2;

using namespace std;

int main(){

    int Mask[NumOfStores][NumOfTypes];
    int store=0,type=0,stop=0;

    cout << "Please input the number of masks for each store:\n";
    for(int i=0;i<NumOfStores;i++){
        for(int j=0;j<NumOfTypes;j++){
            cin >> Mask[i][j];
        }
    }


    while(stop < (NumOfStores*NumOfTypes) ){
        cout << "which store?";
        cin >> store;
        cout << "which type?";
        cin >> type;

        if((store > 0 && store <=NumOfStores) && (store > 0 && type <=NumOfTypes) ){ // 輸入要正確

            if(type==1){ // 成人
                if(Mask[store-1][type-1] >= 3){
                    Mask[store-1][type-1] -= 3;
                    cout << "Masks bought";
                    cout << endl;
                }else{
                    cout << "Out of stock";
                    cout << endl;
                }

            }else{
                if(Mask[store-1][type-1] >= 5){
                    Mask[store-1][type-1] -= 5;
                    cout << "Masks bought";
                    cout << endl;
                }else{
                    cout << "Out of stock";
                    cout << endl;
                }
            }

        }else{
            cout << "Wrong input" << endl;
        }


        stop=0;
        for(int i=0;i<NumOfStores;i++){
            if(Mask[i][0] < 3){
                stop++;
            }
            if(Mask[i][1] < 5){
                stop++;
            }
        }
        //cout << stop;
    }


    cout << "The remaining masks in each store are:" << endl;
    for(int i=0;i<NumOfStores;i++){
        for(int j=0;j<NumOfTypes;j++){
            cout << Mask[i][j] << " ";
            if(j==1 && i!=NumOfStores-1){
                cout << endl;
            }
        }
    }

    return 0;
}
